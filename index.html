Flow of control


$app->get(
    '/launch-nukes',
    function( Request $httpRequest ) use ( $factory ) {
        $request = new LaunchNukesRequest(
            $httpRequest->query->get( 'password' ),
            $httpRequest->query->get( 'number' )
        );

        $r = $factory->newLaunchNukesUseCase()->launchNukes( $request );
        return new Response(
            $factory->newNukeLaunchingResultPresenter()->present( $r )
        );
    }
);
